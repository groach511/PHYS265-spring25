import numpy as np

def gravitational_potential(M, x_M, y_M, x, y):
    G = 6.67430e-11  # Gravitational constant in m^3 kg^-1 s^-2
    r = np.sqrt((x - x_M)**2 + (y - y_M)**2)  # Distance between mass and evaluation point
   
    if r == 0:
        raise ValueError("Singularity encountered: Evaluation point coincides with mass location.")
   
    potential = -G * M / r
    return potential

# usage
M = 5.972e24  # Mass of Earth in kg
x_M, y_M = 0, 0  # Location of mass
x, y = 1e7, 1e7  # evaluation point

phi = gravitational_potential(M, x_M, y_M, x, y)
print(f"Gravitational potential at ({x}, {y}): {phi} J/kg")

# grid range (-1.5d_Earth to +1.5d_Earth)
lim = 1.5 * d_Earth
resolution = 500  # Resolution of the grid
x = np.linspace(-lim, lim, resolution)
y = np.linspace(-lim, lim, resolution)
X, Y = np.meshgrid(x, y)

# gravitational potential
R = np.sqrt(X**2 + Y**2)
Phi = -G * M / R

# singularity at the origin

plt.ylabel('y (m)')
plt.title('Gravitational Potential Around Earth')
plt.axis('equal')

import numpy as np

# Constants from the table
G = 6.67e-11  # Gravitational constant in m^3/kg/s^2

def gravitational_force(M1, m2, x1, y1, x2, y2):
    """
    Compute the gravitational force components (Fx, Fy) exerted by M1 on m2.
   
    Parameters:
    M1 : float - Mass of the first body (kg)
    m2 : float - Mass of the second body (kg)
    x1, y1 : float - Cartesian coordinates of M1 (m)
    x2, y2 : float - Cartesian coordinates of m2 (m)
   
    Returns:
    Fx, Fy : tuple of floats - Force components in the x and y directions (N)
    """
    # Compute displacement vector components
    dx = x2 - x1
    dy = y2 - y1
   
    # Compute the distance with a small softening factor to avoid singularity
    epsilon = 1e-10  # Small value to prevent division by zero
    r = np.sqrt(dx**2 + dy**2) + epsilon
   
    # Compute the force magnitude
    F = G * M1 * m2 / r**2
   
    # Compute the force components
    Fx = F * (dx / r)
    Fy = F * (dy / r)
   
    return Fx, Fy

import numpy as np
import matplotlib.pyplot as plt

# Constants from the table
G = 6.67e-11  # Gravitational constant (m^3/kg/s^2)
M_E = 5.9e24  # Mass of Earth (kg)
M_M = 7.3e22  # Mass of Moon (kg)
d_EM = 3.8e8  # Distance from Earth to Moon (m)
m_CM = 5500  # Mass of Apollo Command Module (kg)

# Moon's position
x_M = -d_EM / np.sqrt(2)
y_M = d_EM / np.sqrt(2)

# Define the grid for plotting
grid_points = 50
x = np.linspace(-1.5 * d_EM, 1.5 * d_EM, grid_points)
y = np.linspace(-1.5 * d_EM, 1.5 * d_EM, grid_points)
X, Y = np.meshgrid(x, y)

# Compute forces due to Earth and Moon
def gravitational_force(M, x_source, y_source, X, Y):
    """Compute the gravitational force components at each grid point."""
    dx = X - x_source
    dy = Y - y_source
    r = np.sqrt(dx**2 + dy**2)
    r3 = r**3 + 1e3  # Avoid division by zero
    Fx = -G * M * m_CM * dx / r3
    Fy = -G * M * m_CM * dy / r3
    return Fx, Fy

# Compute forces
Fx_E, Fy_E = gravitational_force(M_E, 0, 0, X, Y)  # Force due to Earth
Fx_M, Fy_M = gravitational_force(M_M, x_M, y_M, X, Y)  # Force due to Moon

# Total forces
Fx = Fx_E + Fx_M
Fy = Fy_E + Fy_M

# Magnitude of force for color mapping
F_magnitude = np.sqrt(Fx**2 + Fy**2)

# Plot the streamplot
plt.figure(figsize=(10, 8))
strm = plt.streamplot(X, Y, Fx, Fy, color=np.log10(F_magnitude), cmap='plasma', linewidth=1, density=1.2)
cbar = plt.colorbar(label="Log(Force Magnitude)")

# Plot Earth and Moon positions
plt.scatter(0, 0, color='blue', s=100, label="Earth")
plt.scatter(x_M, y_M, color='gray', s=50, label="Moon")

# Labels and title
plt.xlabel("X Position (m)")
plt.ylabel("Y Position (m)")
plt.title("Gravitational Force Field on Apollo 11 Command Module")
plt.legend()
plt.show()

import numpy as np

def delta_v(t, m0, mf, mdot, ve, g):
    # Calculate remaining mass at time t
    m_t = m0 - mdot * t
   
    # Ensure mass doesn't drop below the dry mass (when fuel is gone)
    if m_t < mf:
        return 0
   
    # Calculate change in velocity using the Tsiolkovsky rocket equation
    delta_v = ve * np.log(m0 / m_t) - g * t
    return delta_v

# Example usage values from the provided data
